pragma solidity >= 0.8.0;
 
contract Fundraiser {
    struct Campaign {
        address creator;
        string name;
        uint256 goal;
        uint256 maxDonation;
        string fundType;
        uint etherAmount;
        string thumbnail;
    }
    Campaign[] campaignArr;
 
    // uint public etherCollected = 0 ether;
    address public owner;
    constructor() {
        owner = msg.sender;
    }
 
    modifier onlyOwner(uint index) {
        Campaign memory pulledCampaign = campaignArr[index];
        require(keccak256(abi.encodePacked(pulledCampaign.creator)) == keccak256(abi.encodePacked(msg.sender)));
        _;
    }
 
    modifier notOwner(uint index) {
        Campaign memory pulledCampaign = campaignArr[index];
        Erequire(keccak256(abi.encodePacked(pulledCampaign.creator)) != keccak256(abi.encodePacked(msg.sender)));
        _;
    }
 
    modifier donationWithinBounds(uint index) {
        Campaign memory currCampaign = campaignArr[index];
        Erequire(msg.value <= currCampaign.maxDonation);    
        _;
    }
 
    function makeCampaign(address sender, string memory _name, uint256 _goal, uint256 _maxDonation, string memory _fundType, string memory _thumbnail) public {
        Campaign memory newCampaign = Campaign(sender, _name, _goal / 1000 * 1 ether, _maxDonation  / 1000 * 1 ether, _fundType, 0, _thumbnail);
        campaignArr.push(newCampaign);
    }
 
    function donate(uint index) public payable donationWithinBounds(index) notOwner(index) {
        campaignArr[index].etherAmount += msg.value;
    }
 
    function isGoalReached(uint index) onlyOwner(index) public view returns (bool)  {
        Campaign memory pulledCampaign = campaignArr[index];
        return pulledCampaign.etherAmount >= pulledCampaign.goal;
    }
 
    function endFundraiser(uint index) public payable onlyOwner(index) {
        require(isGoalReached(index) == true);
        Campaign memory pulledCampaign = campaignArr[index];
        payable(pulledCampaign.creator).transfer(pulledCampaign.etherAmount);
    }
 
    function getList() public view returns (Campaign[] memory) {
        return campaignArr;
    }
 
    function getEther(uint index) public view returns (uint) {
        Campaign memory pulledCampaign = campaignArr[index];
        return pulledCampaign.etherAmount / 1E18;
    }
 
    function getCampaign(uint index) public view returns (Campaign memory) {
        return campaignArr[index];
    }
